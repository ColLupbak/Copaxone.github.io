/* ---------- rotation math ---------- */
function countWeekdayOccurrences(N, startDOW, targetDOW){
  if(N<=0) return 0;
  const offset = (((targetDOW - startDOW) % 7) + 7) % 7;
  if(offset >= N) return 0;
  return Math.floor((N - 1 - offset)/7) + 1;
}
// number of injection weekdays in the inclusive range [a, b] (requires a <= b)
function countInclusive(a, b){
  const N = daysBetween(a, b) + 1;
  let cnt = 0;
  for(const w of INJ_DOW) cnt += countWeekdayOccurrences(N, a.getDay(), w);
  return cnt;
}
// sequence number of an injection day relative to the start (start = 0), works both directions
function seqFor(dateKey){
  const start = fromKey(settings.startKey);
  const target = fromKey(dateKey);
  if(daysBetween(start, target) >= 0) return countInclusive(start, target) - 1;
  return -(countInclusive(target, start) - 1);
}
function plannedSite(dateKey){
  const s = seqFor(dateKey);
  return ((settings.startSiteIndex + s) % 7 + 7) % 7;
}

/* status of an injection day: taken | skipped | due | missed | upcoming */
function statusFor(dateKey){
  const entry = log[dateKey];
  if(entry && entry.status) return entry.status;
  const diff = daysBetween(new Date(), fromKey(dateKey)); // negative=past, 0=today, positive=future
  if(diff === 0) return "due";
  if(diff < 0) return "missed";
  return "upcoming";
}
function siteFor(dateKey){
  const entry = log[dateKey];
  if(entry && Number.isInteger(entry.site)) return entry.site;
  return plannedSite(dateKey);
}

  if(!isInjDow(today)) return;
  const k = todayKey();

  if(meta.lastNotified === k) return;      // already notified today
  const [h,m] = (settings.remTime||"09:00").split(":").map(Number);
  if(today.getHours()<h || (today.getHours()===h && today.getMinutes()<m)) return;
  const site = SITES[siteFor(k)];
  try{
    new Notification("Copaxone injection due today", { body:`Today's site: ${site}. Tap the app to confirm once taken.` });
  }catch(e){}
  meta.lastNotified = k; save();
}
